let Magix = require('magix');
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            page: 1,
            size: 40
        });
    },
    'change<change>' (e) {
        // e.page 当前第几页
        // e.size 每页多少条
        // e.offset 偏移量
        this.updater.digest({
            page: e.page,
            size: e.size
        })
    }
});
