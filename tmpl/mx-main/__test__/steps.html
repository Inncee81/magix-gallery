2. 子view必须实现的check方法说明，返回一个Promise
check() {
    let ok = true,
        remain = {};

    return new Promise((resolve) => {
        // 此处返回promise，防止有接口提交校验等
        resolve({
            ok,  // 是否校验通过，false的时候不提交
            msg: '错误信息提示',
            remain: {} // 当前步骤保留的参数，所有子view交集
        })
    })
}
