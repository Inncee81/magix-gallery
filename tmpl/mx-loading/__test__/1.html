let Magix = require('magix');
let Loading = require('@../index'); //loading组件
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest();
    },
    'show<click>'(e){
        Loading.showLoading();

        setTimeout(() => {
            Loading.hideLoading();
        }, 3000);
    }
});
