以下示例：
手动打开loading进度条
进度条存在时依然可以操作页面
打开loading 关闭loading
HTML Code
{{!text1}}  
<a href="javascript:;" class="btn btn-brand mr20"
    mx-click="show()" >打开</a> 

<a href="javascript:;" class="btn"
    mx-click="hide()" >关闭</a>
JS Code
{{!text2}}  
let Magix = require('magix');
let Loading = require('@../../mx-loading/index'); //loading组件
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    mixins: [Loading],
    render() {
        this.updater.digest();
    },
    'show<click>'(e){
        // showLoading(closeMask)
        // closeMask不需要禁止操作
        this.showLoading(true);
    },
    'hide<click>'(e){
        this.hideLoading();
    }
});
