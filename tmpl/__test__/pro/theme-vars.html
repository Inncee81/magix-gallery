在本地项目 project/src/project/assets/vars_override.less 文件中覆盖可配置参数，可配置参数如下所示：
// 品牌色
@color-brand: #4d7fff;   //品牌色
@color-brand-hover: shade(@color-brand, 8%); //品牌色hover时色值
@color-brand-vs: #14c9ce;   //品牌色对比色
@color-brand-text: #ffffff;     //品牌色上文案色值
@color-brand-text-hover: #ffffff;     //hover时品牌色上文案色值
@color-brand-light : tint(@color-brand, 78%);
@color-brand-opacity: tint(@color-brand, 95%);

//边框
@color-border: #e6e6e6;
@border-radius: 4px;
@border-highlight: #e6e6e6;
@border-highlight-hover: #ccc;

// 表格相关色值
@table-color-border: #ebebeb;
@table-operation-color-bg: tint(@color-brand, 88%);

// 品牌按钮颜色，支持渐变，若不需要渐变，@btn-brand = @btn-brand-gradient 即可
@btn-border-radius: @border-radius;
@btn-brand: @color-brand;
@btn-brand-gradient: @color-brand;
@btn-brand-hover: @color-brand-hover;
@btn-brand-gradient-hover: @color-brand-hover;
@btn-brand-text: @color-brand-text;
@btn-brand-text-hover: @color-brand-text-hover;

// 普通默认按钮文字颜色， 边框颜色，背景色
@btn-text: #333;
@btn-text-hover: #333;
@btn-border: #e6e6e6;
@btn-border-hover: #ccc;
@btn-bg: #e6e6e6;
@btn-bg-hover: #ccc;

//字体相关参数
@line-height: 1.5;
@font-size: 12px;

// 提示颜色
@color-primary: #333;
@color-secondary: #666;
@color-third: #999;
@color-hover: #f0f0f0;
@color-warn: #ffb400;
@color-red: #a40100;
@color-orange: #ff5500;
@color-green: #51a300;
@color-blue: #4d7fff;
@color-disabled: #eee;

//背景色
@color-bg: #fafafa;
@color-bg-hover: #ccc;

//输入或下拉框高度 
@input-height: 32px;
@input-small-height: 26px;

//动画时长
@duration: 0.25s;
