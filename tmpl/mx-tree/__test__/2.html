使用场景：
实时反馈选中结果，每次操作都响应事件
默认选中某些选项
可展开收起子选项
当前已选中： {{=selected.length}}
HTML Code
{{!text1}}  
<mx-tree id="{{=viewId}}_tree"
    list="{{@list}}"
    bottom-values="{{@selected}}"
    mx-change="change()"></mx-tree>
JS Code
{{!text2}}  
let Magix = require('magix');
let Vframe = Magix.Vframe;
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let list = [{
            value: 1,
            pValue: '',
            text: '1'
        }, {
            value: 2,
            pValue: '',
            text: '2'
        }, {
            value: 3,
            pValue: '',
            text: '3'
        }, {
            value: 11,
            pValue: 1,
            text: '1-1'
        }, {
            value: 12,
            pValue: 1,
            text: '1-2'
        }, {
            value: 21,
            pValue: 2,
            text: '2-1'
        }, {
            value: 211,
            pValue: 21,
            text: '2-1-1'
        }, {
            value: 212,
            pValue: 21,
            text: '2-1-2'
        }, {
            value: 22,
            pValue: 2,
            text: '2-2'
        }, {
            value: 23,
            pValue: 2,
            text: '2-3'
        }]

        this.updater.digest({
            viewId: this.id,
            list,
            selected: [211, 23, 3]
        });
    },
    'change<change>'(e) {
        let tree = Vframe.get(this.id + '_tree');

        this.updater.digest({
            selected: tree.invoke('getBottomValues')
        })
    }
});
