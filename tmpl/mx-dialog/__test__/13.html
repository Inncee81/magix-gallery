规范淘宝登录框的打开样式：
mixins到view上，this.mxLoginView()
有mask，宽度350，高度400（新版mini高度必须大于=400，否则验证码，加验场景展示不全，查看说明 ），居中显示
使用说明：
直接传入bizCode即可，具体逻辑已在组件中包装，目前已有的bizCode定义如下
项目 bizCode 登陆框样式版本 查看示例
直通车 subway 直接使用默认的淘宝橙即可
无需要单独申请		 打开直通车登录框
钻展 zszw zszwsite_mm 打开钻展登录框
超级推荐 feedFlow dingxiang 打开超级推荐登录框
超级互动 interactive dingxiang 打开超级互动登录框
策略中心 adStrategy dingxiang 打开策略中心登录框
达摩盘服务商版本 dmpIsv mdmp 打开达摩盘服务商版本登录框
联盟 union 直接使用默认的淘宝橙即可 打开联盟登录框
品销宝 brand dingxiang 打开品销宝登录框
默认 zszwsite_mm 打开默认登录框
HTML Code
{{!text1}}  
<a href="javascript:;" mx-click="show()" 
    class="btn btn-brand">打开默认登录框</a>
JS Code
{{!text2}}  
let Magix = require('magix');
let Dialog = require('@../index'); //mixins dialog

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    mixins: [Dialog],
    render() {
        this.updater.digest();
    },
    'show<click>'(e) {
        // this.mxLoginView(viewOptions)
        // viewOptions：object
        //      bizCode：项目bizCode，包装了登陆框逻辑
        this.mxLoginView({
            bizCode: 'adStrategy'
        })
    }
});
