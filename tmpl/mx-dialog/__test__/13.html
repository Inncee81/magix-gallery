规范淘宝登录框的打开样式：
mixins到view上，this.mxLoginView()
有mask，宽度350，高度400（新版mini高度必须大于=400，否则验证码，加验场景展示不全，查看说明 ），居中显示
打开登录框
HTML Code
{{!text1}}  
<a href="javascript:;" mx-click="show()" 
    class="btn btn-brand">打开登录框</a>
JS Code
{{!text2}}  
let Magix = require('magix');
let Dialog = require('@../index'); //mixins dialog

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    mixins: [Dialog],
    render() {
        this.updater.digest();
    },
    'show<click>'(e) {
        // this.mxLoginView(viewPath[, viewOptions])
        // viewPath：登录框view
        // viewOptions：object，带入viewPath的参数
        this.mxLoginView('@./login')
    }
});
