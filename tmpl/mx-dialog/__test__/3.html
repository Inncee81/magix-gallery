 将dialog的方法mixins到view上
可打开多个浮层
HTML Code
{{!text1}}  
<a href="javascript:;" mx-click="open()" 
    class="btn btn-brand">可打开多个浮层</a>
JS Code
{{!text2}}  
let Magix = require('magix');
let Dialog = require('@../index'); //mixins dialog
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    mixins: [Dialog],
    render() {
        this.updater.digest();
    },
    'open<click>'(e) {
        //  viewPath：dialog view路径
        //  viewOptions：传入dialog的数据，挂载当前dialog实体
        //  dialogOptions：样式覆盖
        //      width： 宽度
        //      height： 高度
        //      modal：是否允许滚动
        //      mask：是否有遮罩
        //      ......
        this.mxDialog('@./multi', {
            number: 1
        }, {
            width: 800,
            height: 270,
            modal: true
        });
    }
});
