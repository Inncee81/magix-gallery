let Magix = require('magix');
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            disabled: true,
            checked: true
        });
    },
    'outerChange<click>'() {
        let disabled = this.updater.get('disabled'),
            checked = this.updater.get('checked');
        this.updater.digest({
            disabled: !disabled,
            checked: !checked
        });
    },
    'innerChange<change>'(e){
        this.updater.digest({
            checked: e.target.checked
        })
    }
});
