JS Code
{{!text2}}  
let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@trigger.html',
    render() {
        this.updater.digest({
            open: false
        });
    },
    'toggle<click>'(){
        let open = this.updater.get('open');
        this.updater.digest({
            open: !open
        })
    }
});
