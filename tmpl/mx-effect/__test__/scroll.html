mx-effect scroll

HTML Code:
<mx-effect.scroll value="<%@value%>" />
<button type="button" mx-click="update()" class="btn btn-brand ml20">update</button>
Javascript Code:
let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@scroll.html',
    render() {
        this.updater.digest({
            value: 100
        });
        setInterval(() => {
            let v = (Math.random() * 20) + this.updater.get('value');
            this.updater.digest({
                value: v
            });
        }, 1000);
    },
    'update<click>'() {
        this.updater.digest({
            value: this.updater.get('value') - 20
        });
    }
});
