let Magix = require('magix');
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let list = [{
            value: 1,
            text: '测试Logo'
        }, {
            value: 2,
            text: 'logo'
        }, {
            value: 3,
            text: '123456'
        }];
        this.updater.digest({
            list: list,
            selectedValue: list[1].value
        });
    },
    'select<change>' (e) {
        // e.value 当前选中值
        // e.text 当前选中文案
        this.updater.digest({
            selectedValue: e.value
        });
    }
});
