let Magix = require('magix');
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest();
    },
    'change<click>'(e){
        let that = this;
        that.updater.digest({
            loading: true
        })
        setTimeout(() => {
            that.updater.digest({
                loading: false
            })
        }, 2000)
    }
});
